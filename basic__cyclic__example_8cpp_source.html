1// examples/basic_cyclic_example.cpp
2// Minimal, bare-metal-friendly example showing how to use the cyclic executive.
3// - Static devices
4// - Static schedule (minor frames)
5// - Static ExecContext
6// - Fake time source driving the executor
7//
8// This file intentionally avoids C++ iostreams and dynamic allocation so it
9// can serve as documentation-by-example for embedded use.

24// Fake monotonic time source (microseconds)
25static uint64_t g_time_us = 0;
26static uint64_t fake_time_source() { return g_time_us; }
27
28// Simple device-side state used in the example to show init/update were called.
29struct DevState { uint32_t inits; uint32_t updates; };
30
31static DevState dev0_state = {0,0};
32static DevState dev1_state = {0,0};
33
34// Device callbacks (no allocations)
35static void dev0_init(Device* d) { (void)d; dev0_state.inits++; }
36static void dev0_update(Device* d) { (void)d; dev0_state.updates++; }
37
38static void dev1_init(Device* d) { (void)d; dev1_state.inits++; }
39static void dev1_update(Device* d) { (void)d; dev1_state.updates++; }
40
41// Static device table (caller-owned, remains valid for lifetime)
42static Device devices[] = {
43 // init, update, period_ms, wcet_us, start, stop, on_fault
44 { dev0_init, dev0_update, 1000, 1000, nullptr, nullptr, nullptr },
45 { dev1_init, dev1_update, 1000, 500, nullptr, nullptr, nullptr }
46};
47
48// Minor frames: each minor references device indices in deterministic order.
49static const uint16_t minor0_devs[] = { 0, 1 }; // execute dev0 then dev1
50static const uint16_t minor1_devs[] = { 1 }; // execute dev1 only
51
52static const MinorFrameDef minors[] = {
53 // device_indices, device_count, duration_us
54 { minor0_devs, 2, 100000 }, // 100 ms
55 { minor1_devs, 1, 200000 } // 200 ms
56};
57
58static const ScheduleDef schedules[] = {
59 { minors, 2 }
60};
61
62// Statically-allocated runtime buffers (caller-provided)
63static Health health[sizeof(devices)/sizeof(devices[0])];
64static DeviceState states[sizeof(devices)/sizeof(devices[0])];
65static ExecContext ctx; // context owned by caller
66
67int main_example(void) {
68 // Initialize the cyclic executor with our static context and tables.
69 bool ok = cyclic_init(&ctx,
70 devices,
71 sizeof(devices)/sizeof(devices[0]),
72 schedules,
73 1,
74 0, // initial schedule index
75 health,
76 states,
77 fake_time_source,
78 OverrunPolicy::DropTask);
79 if (!ok) {
80 // In embedded use, handle init failure appropriately (here we just return)
81 return -1;
82 }
83
84 // Drive the executor by advancing fake time. At each tick we may execute
85 // zero or one minor frames depending on time progression. The example
86 // advances time in 100 ms steps and polls the executor.
87
88 for (int step = 0; step < 6; ++step) {
89 // Advance time by 100 ms
90 g_time_us += 100000;
91 // Poll the executor; this will run the minor frame(s) whose deadline passed
92 cyclic_poll(&ctx);
93
94 // Example visibility: print counters using printf (non-iostream)
95 // On a bare-metal system remove these prints and observe device-side state.
96 std::printf("time=%llu us: dev0 inits=%u updates=%u; dev1 inits=%u updates=%u\n",
97 (unsigned long long)g_time_us,
98 (unsigned)dev0_state.inits, (unsigned)dev0_state.updates,
99 (unsigned)dev1_state.inits, (unsigned)dev1_state.updates);
100 }
101
102 return 0;
103}
