47static void test_init_and_basic_execution() {
48 // device counters
49 static Counters A{0,0};
50 static Counters B{0,0};
51
52 // lambdas cannot be converted to function pointers; provide static functions
53 struct D0 { static void init(Device*) { A.inits++; } static void update(Device*) { A.updates++; } };
54 struct D1 { static void init(Device*) { B.inits++; } static void update(Device*) { B.updates++; } };
55
56 static Device devices[] = {
57 { &D0::init, &D0::update, 1000, 200, nullptr, nullptr, nullptr },
58 { &D1::init, &D1::update, 1000, 300, nullptr, nullptr, nullptr }
59 };
60
61 static const uint16_t minor0[] = { 0 };
62 static const uint16_t minor1[] = { 1 };
63 static const MinorFrameDef minors[] = { { minor0, 1, 100000 }, { minor1, 1, 100000 } };
64 static const ScheduleDef schedules[] = { { minors, 2 } };
65
66 static Health health[2];
67 static DeviceState states[2];
68 ExecContext ctx{};
69
70 g_time_us = 0;
71 bool ok = cyclic_init(&ctx, devices, 2, schedules, 1, 0, health, states, fake_time_source, OverrunPolicy::DropTask);
72 ASSERT_EQ(true, ok);
73 // init should have been called for both devices
74 ASSERT_EQ(1, A.inits);
75 ASSERT_EQ(1, B.inits);
76
77 // advance time to trigger first minor
78 g_time_us += 100000; // minor 0 deadline
79 cyclic_tick_us(&ctx, g_time_us);
80 // device 0 update executed
81 ASSERT_EQ(1, A.updates);
82 ASSERT_EQ(0, B.updates);
83
84 // advance to next minor
85 g_time_us += 100000;
86 cyclic_tick_us(&ctx, g_time_us);
87 ASSERT_EQ(1, B.updates);
88}
