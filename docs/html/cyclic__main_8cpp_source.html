2// Clean example showing how to use the modular cyclic executive API.
3// - Uses only modular headers (core/, schedule/, executive/, policy/)
4// - Bare-metal-friendly (uses printf for visibility)

17using namespace dfw;
18
19// Fake monotonic time source (microseconds). In real embedded systems provide
20// a hardware timer or OS monotonic clock.
21static uint64_t g_time_us = 0;
22static uint64_t fake_time_source() { return g_time_us; }
23
24// Simple device-local counters to demonstrate init/update being called.
25struct Counters { uint32_t inits; uint32_t updates; };
26static Counters devA_c = {0,0};
27static Counters devB_c = {0,0};
28
29// Device callbacks (no dynamic allocation)
30static void devA_init(Device* d) { (void)d; devA_c.inits++; }
31static void devA_update(Device* d) { (void)d; devA_c.updates++; }
32
33static void devB_init(Device* d) { (void)d; devB_c.inits++; }
34static void devB_update(Device* d) { (void)d; devB_c.updates++; }
35
36// Static device table (caller-owned)
37static Device devices[] = {
38 { devA_init, devA_update, 1000, 200, nullptr, nullptr, nullptr },
39 { devB_init, devB_update, 1000, 300, nullptr, nullptr, nullptr }
40};
41
42// Minor frames (deterministic ordered device indices)
43static const uint16_t minor0_devs[] = { 0 }; // minor 0: run device 0
44static const uint16_t minor1_devs[] = { 1 }; // minor 1: run device 1
45
46static const MinorFrameDef minors[] = {
47 { minor0_devs, 1, 100000 }, // 100 ms
48 { minor1_devs, 1, 100000 } // 100 ms
49};
50
51static const ScheduleDef schedules[] = { { minors, 2 } };
52
53// Caller-provided runtime buffers (statically allocated)
54static Health health[sizeof(devices)/sizeof(devices[0])];
55static DeviceState states[sizeof(devices)/sizeof(devices[0])];
56static ExecContext ctx;
57
58int main() {
59 // Initialize the cyclic executor using our static context and tables.
60 bool ok = cyclic_init(&ctx,
61 devices,
62 sizeof(devices)/sizeof(devices[0]),
63 schedules,
64 1, // schedule_count
65 0, // initial schedule index
66 health,
67 states,
68 fake_time_source,
69 OverrunPolicy::DropTask);
70 if (!ok) return 1;
71
72 // Drive the executor: advance time and poll. Each poll may execute completed
73 // minor frames deterministically based on the fake time source.
74 for (int i = 0; i < 6; ++i) {
75 g_time_us += 100000; // advance by 100 ms
76 cyclic_poll(&ctx);
77
78 // Print counters to show where init/update ran.
79 // On a constrained target replace printf with suitable telemetry.
80 std::printf("t=%lluus: A inits=%u updates=%u | B inits=%u updates=%u\n",
81 (unsigned long long)g_time_us,
82 (unsigned)devA_c.inits, (unsigned)devA_c.updates,
83 (unsigned)devB_c.inits, (unsigned)devB_c.updates);
84 }
85
86 return 0;
87}
