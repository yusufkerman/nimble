27 // ------------ Legacy periodic mode (caller provides next_release buffer) ------------
28 // Initializes legacy periodic (next-release) mode. All time is kept
29 // internally in microseconds to avoid truncation. The caller must
30 // provide a preallocated `next_release_buffer` with length >= count.
31 static bool init(Device* devices, std::size_t count, uint32_t major_frame_ms,
32 uint64_t* next_release_buffer,
33 CatchUpPolicy policy = CatchUpPolicy::Limited,
34 uint32_t max_catchups = 1) {
35 if (!devices || !next_release_buffer) return false;
36 devices_ = devices;
37 count_ = count;
38 // informational only
39 major_frame_ms_ = major_frame_ms;
40
41 // prepare legacy state
42 next_release_ = next_release_buffer;
43 policy_ = policy;
44 max_catchups_ = max_catchups;
45 current_time_us_ = 0;
46 for (std::size_t i = 0; i < count_; ++i) {
47 if (devices_[i].init) devices_[i].init(&devices_[i]);
48 next_release_[i] = (devices_[i].period_ms == 0) ? kNever
49 : static_cast<uint64_t>(devices_[i].period_ms) * 1000ull;
50 }
51 legacy_mode_ = true;
52 cyclic_mode_ = false;
53 return true;
54 }

57 // be reloaded for legacy mode.

59 current_time_us_ = 0;

62 next_release_[i] = (devices_[i].period_ms == 0) ? kNever
63 : static_cast<uint64_t>(devices_[i].period_ms) * 1000ull;
64 }
65 }
66 minor_elapsed_us_ = 0;
67 current_minor_ = 0;
68 skip_next_minor_ = false;

92 // Initialize cyclic scheduling with static minor frame table.
93 // The caller must provide:
94 // - `health_buffer` : array of Health sized `device_count`
95 // - `state_buffer` : array of DeviceState sized `device_count`
96 // Both buffers must remain valid for the lifetime of the executive.
97 // Returns false on invalid indices or schedulability failure.
98 using DeviceFaultHandlerFn = void(*)(std::size_t device_index, FaultReason reason);
99 static bool init_cyclic(Device* devices, std::size_t device_count,
100 const MinorFrameDef* minors, std::size_t minor_count,
101 Health* health_buffer, DeviceState* state_buffer,
102 OverrunPolicy overrun_policy = OverrunPolicy::DropTask,
103 FaultHandlerFn frame_fault_handler = nullptr,
104 DeviceFaultHandlerFn device_fault_handler = nullptr,
105 ResetHookFn reset_hook = nullptr,
106 TimeSourceFn time_source = nullptr,
107 uint32_t overrun_threshold = 1) {
108 if (!devices || !minors || minor_count == 0) return false;
109 if (!health_buffer || !state_buffer) return false;
110 devices_ = devices;
111 count_ = device_count;
112 minors_ = minors;
113 minor_count_ = minor_count;
114 overrun_policy_ = overrun_policy;
115 frame_fault_handler_ = frame_fault_handler;
116 device_fault_handler_ = device_fault_handler;
117 reset_hook_ = reset_hook;
118 time_source_ = time_source;
119 health_ = health_buffer;
120 states_ = state_buffer;
121 overrun_threshold_ = overrun_threshold;
122
123 // static schedulability check: sum WCET per minor must fit its budget
124 major_frame_us_ = 0;
125 for (std::size_t m = 0; m < minor_count_; ++m) {
126 major_frame_us_ += static_cast<uint64_t>(minors_[m].duration_us);
127 uint64_t sum_wcet = 0;
128 for (std::size_t j = 0; j < minors_[m].device_count; ++j) {
129 uint16_t di = minors_[m].device_indices[j];
130 if (di >= count_) return false; // bad table
131 sum_wcet += devices_[di].wcet_us; // wcet in us
132 if (sum_wcet > minors_[m].duration_us) return false; // not schedulable
133 }
134 }
135
136 // init cyclic state
137 cyclic_mode_ = true;
138 legacy_mode_ = false;
139 current_minor_ = 0;
140 minor_elapsed_us_ = 0;
141 last_time_us_ = time_source_ ? time_source_() : 0;
142 skip_next_minor_ = false;
143
144 // Initialize health and state buffers deterministically
145 for (std::size_t i = 0; i < count_; ++i) {
146 health_[i].execution_count = 0;
147 health_[i].overrun_count = 0;
148 health_[i].last_exec_time_us = 0;
149 health_[i].max_exec_time_us = 0;
150 states_[i] = DeviceState::Initialized;
151 }
152 return true;
153 }
154 }

156 // ------------ Tick helpers ------------
157 // tick accepts milliseconds (common) and forwards to microsecond version
158 static void tick(uint32_t delta_ms) {
159 if (cyclic_mode_) {
160 tick_us(static_cast<uint64_t>(delta_ms) * 1000ull);
161 return;
162 }
163 // legacy periodic mode
164 if (!legacy_mode_ || !devices_ || count_ == 0 || !next_release_) return;
165 uint64_t delta_us = static_cast<uint64_t>(delta_ms) * 1000ull;
166 current_time_us_ += delta_us;
167
168 for (std::size_t i = 0; i < count_; ++i) {
169 Device& d = devices_[i];
170 if (!d.update || d.period_ms == 0) continue;
171 if (current_time_us_ < next_release_[i]) continue;
172
173 uint64_t period_us = static_cast<uint64_t>(d.period_ms) * 1000ull;
174 uint64_t diff = current_time_us_ - next_release_[i];
175 uint64_t total_due = diff / period_us + 1;
176
177 // execute according to policy
178 if (policy_ == CatchUpPolicy::CatchAll) {
179 for (uint64_t k = 0; k < total_due; ++k) d.update(&d);
180 } else if (policy_ == CatchUpPolicy::Limited) {
181 uint64_t calls = total_due;
182 if (calls > max_catchups_) calls = max_catchups_;
183 for (uint64_t k = 0; k < calls; ++k) d.update(&d);
184 }
185 // SkipMisses => don't call missed activations
186
187 // advance next_release_ with saturation
188 advance_next_release_saturating(next_release_[i], period_us, total_due);
189 }
190 }

192 // tick_us: microsecond-resolution tick used by cyclic-mode or time-source
193 static void tick_us(uint64_t delta_us) {
194 if (!cyclic_mode_ || !minors_) return;
195 minor_elapsed_us_ += delta_us;
196
197 // while we've completed current minor frame, execute it deterministically
198 while (minor_elapsed_us_ >= minors_[current_minor_].duration_us) {
199 const MinorFrameDef& mf = minors_[current_minor_];
200 uint64_t frame_budget = mf.duration_us;
201 uint64_t used_us = 0;
202 bool overrun = false;
203
204 for (std::size_t idx = 0; idx < mf.device_count; ++idx) {
205 std::size_t dev_idx = mf.device_indices[idx];
206 uint64_t wcet = devices_[dev_idx].wcet_us;
207
208 // account the declared WCET deterministically
209 used_us += wcet;
210 if (used_us > frame_budget) { overrun = true; /* stop accounting */ break; }
211
212 // execute the device's update (application work)
213 if (devices_[dev_idx].update) devices_[dev_idx].update(&devices_[dev_idx]);
214 }
215
216 if (overrun) {
217 // deterministic handling according to configured policy
218 handle_overrun(current_minor_, used_us, static_cast<uint32_t>(frame_budget));
219 }
220
221 // advance minor frame timeline
222 minor_elapsed_us_ -= mf.duration_us;
223 current_minor_ = (current_minor_ + 1) % minor_count_;
224
225 if (skip_next_minor_) {
226 // skip execution of the next minor frame (deterministic)
227 skip_next_minor_ = false;
228 minor_elapsed_us_ = 0;
229 current_minor_ = (current_minor_ + 1) % minor_count_;
230 }
231 }
232 }

234 // poll_from_time_source: call when a TimeSourceFn was provided in init_cyclic
235 static void poll_from_time_source() {
236 if (!time_source_ || !cyclic_mode_) return;
237 uint64_t now = time_source_();
238 uint64_t delta = now - last_time_us_;
239 last_time_us_ = now;
240 tick_us(delta);
241 }

243private:
244 // small helper: saturating advance of next_release by (period_us * times)
245 static inline void advance_next_release_saturating(uint64_t &next, uint64_t period_us, uint64_t times) {
246 if (next == kNever) return;
247 if (period_us == 0) { next = kNever; return; }
248 if (times == 0) return;
249 if (times > UINT64_MAX / period_us) { next = kNever; return; }
250 uint64_t incr = times * period_us;
251 if (next > UINT64_MAX - incr) { next = kNever; return; }
252 next += incr;
253 }

255 // overrun handling helper
256 static inline void handle_overrun(std::size_t minor_idx, uint32_t used_us, uint32_t budget_us) {
257 switch (overrun_policy_) {
258 case OverrunPolicy::DropTask:
259 // Do nothing further: remaining tasks in this minor are not executed
260 break;
261 case OverrunPolicy::SkipFrame:
262 skip_next_minor_ = true;
263 break;
264 case OverrunPolicy::SignalFault:
265 if (fault_handler_) fault_handler_(minor_idx, /*device_index*/ 0, used_us, budget_us);
266 break;
267 case OverrunPolicy::ResetSystem:
268 if (reset_hook_) reset_hook_();
269 break;
270 }
271 }

385 // Initialize cyclic-mode with caller-provided minor-frame table. All
386 // buffers must be statically allocated by the caller. This function
387 // performs static schedulability checks (sum WCETs <= minor duration).
388 static bool init_cyclic(Device* devices, std::size_t device_count,
389 const MinorFrameDef* minors, std::size_t minor_count,
390 OverrunPolicy overrun_policy = OverrunPolicy::DropTask,
391 FaultHandlerFn fault_handler = nullptr,
392 ResetHookFn reset_hook = nullptr,
393 TimeSourceFn time_source = nullptr) {
394 if (!devices || !minors || minor_count == 0) return false;
395 // assign
396 devices_ = devices;
397 count_ = device_count;
398 minors_ = minors;
399 minor_count_ = minor_count;
400 overrun_policy_ = overrun_policy;
401 fault_handler_ = fault_handler;
402 reset_hook_ = reset_hook;
403 time_source_ = time_source;
404
405 // compute major frame and validate per-minor WCET sums (static check)
406 major_frame_us_ = 0;
407 for (std::size_t m = 0; m < minor_count_; ++m) {
408 major_frame_us_ += static_cast<uint64_t>(minors_[m].duration_us);
409 uint64_t sum_wcet = 0;
410 for (std::size_t j = 0; j < minors_[m].device_count; ++j) {
411 uint16_t di = minors_[m].device_indices[j];
412 if (di >= count_) return false; // invalid index
413 sum_wcet += devices_[di].wcet_us;
414 if (sum_wcet > minors_[m].duration_us) return false; // schedulability fail
415 }
416 }
417
418 // initialize cyclic state
419 cyclic_mode_ = true;
420 legacy_mode_ = false;
421 current_minor_ = 0;
422 minor_elapsed_us_ = 0;
423 last_time_us_ = time_source_ ? time_source_() : 0;
424 skip_next_minor_ = false;
425 return true;
426 }

428 // Advance the executive by `delta_ms` (milliseconds). This will
429 // dispatch either legacy periodic mode or cyclic-mode processing.
430 static void tick(uint32_t delta_ms) {
431 if (cyclic_mode_) {
432 tick_us(static_cast<uint64_t>(delta_ms) * 1000ull);
433 return;
434 }
435
436 if (!legacy_mode_) return;
437 if (!devices_ || count_ == 0 || !next_release_) return;
438
439 // convert delta to microseconds for internal consistency
440 uint64_t delta_us = static_cast<uint64_t>(delta_ms) * 1000ull;
441 current_time_us_ += delta_us;
442
443 for (std::size_t i = 0; i < count_; ++i) {
444 Device& d = devices_[i];
445 if (!d.update) continue;
446 if (d.period_ms == 0) continue;
447
448 if (current_time_us_ < next_release_[i]) continue;
449
450 uint64_t diff = current_time_us_ - next_release_[i];
451 uint64_t total_due = diff / (static_cast<uint64_t>(d.period_ms) * 1000ull) + 1;
452
453 switch (policy_) {
454 case CatchUpPolicy::CatchAll: {
455 for (uint64_t k = 0; k < total_due; ++k) d.update(&d);
456 // saturating add
457 if (d.period_ms == 0) break;
458 uint64_t mul = static_cast<uint64_t>(d.period_ms) * 1000ull;
459 if (total_due > UINT64_MAX / mul) next_release_[i] = kNever;
460 else {
461 uint64_t incr = total_due * mul;
462 if (next_release_[i] > UINT64_MAX - incr) next_release_[i] = kNever;
463 else next_release_[i] += incr;
464 }
465 break;
466 }
467 case CatchUpPolicy::Limited: {
468 uint64_t calls = total_due;
469 if (calls > max_catchups_) calls = max_catchups_;
470 for (uint64_t k = 0; k < calls; ++k) d.update(&d);
471 if (d.period_ms == 0) break;
472 uint64_t mul = static_cast<uint64_t>(d.period_ms) * 1000ull;
473 if (total_due > UINT64_MAX / mul) next_release_[i] = kNever;
474 else {
475 uint64_t incr = total_due * mul;
476 if (next_release_[i] > UINT64_MAX - incr) next_release_[i] = kNever;
477 else next_release_[i] += incr;
478 }
479 break;
480 }
481 case CatchUpPolicy::SkipMisses: {
482 if (d.period_ms == 0) break;
483 uint64_t mul = static_cast<uint64_t>(d.period_ms) * 1000ull;
484 if (total_due > UINT64_MAX / mul) next_release_[i] = kNever;
485 else {
486 uint64_t incr = total_due * mul;
487 if (next_release_[i] > UINT64_MAX - incr) next_release_[i] = kNever;
488 else next_release_[i] += incr;
489 }
490 break;
491 }
492 }
493 }

496 // tick in microseconds - used for cyclic mode
497 static void tick_us(uint64_t delta_us) {
498 if (!cyclic_mode_ || !minors_) return;
499
500 minor_elapsed_us_ += delta_us;
501
502 // process expired minor frames in order
503 while (minor_elapsed_us_ >= minors_[current_minor_].duration_us) {
504 uint32_t frame_budget = minors_[current_minor_].duration_us;
505 uint32_t used_us = 0;
506
507 const uint16_t* devs = minors_[current_minor_].device_indices;
508 for (std::size_t di = 0; di < minors_[current_minor_].device_count; ++di) {
509 std::size_t dev_idx = devs[di];
510 uint32_t w = devices_[dev_idx].wcet_us;
511 // account WCET deterministically
512 used_us += w;
513
514 if (used_us > frame_budget) {
515 // deterministic overrun handling
516 switch (overrun_policy_) {
517 case OverrunPolicy::DropTask:
518 goto advance_minor;
519 case OverrunPolicy::SkipFrame:
520 skip_next_minor_ = true;
521 goto advance_minor;
522 case OverrunPolicy::SignalFault:
523 if (fault_handler_) fault_handler_(current_minor_, dev_idx, used_us, frame_budget);
524 goto advance_minor;
525 case OverrunPolicy::ResetSystem:
526 if (reset_hook_) reset_hook_();
527 goto advance_minor;
528 }
529 }
530
531 // execute device update (application work)
532 if (devices_[dev_idx].update) devices_[dev_idx].update(&devices_[dev_idx]);
533 }
534
535 advance_minor:
536 // consume minor frame time and advance
537 minor_elapsed_us_ -= frame_budget;
538 current_minor_ = (current_minor_ + 1) % minor_count_;
539 if (skip_next_minor_) {
540 // deterministically skip next minor frame execution
541 skip_next_minor_ = false;
542 // skip its execution and reset elapsed
543 minor_elapsed_us_ = 0;
544 current_minor_ = (current_minor_ + 1) % minor_count_;
545 }
546 }
547 }

554 uint64_t now = time_source_();
555 uint64_t delta = now - last_time_us_;

609// Improved CyclicExecutive: uses per-device next-release times (absolute
610// timeline) so late ticks or variable delta_ms do not cause missed
611// activations. Major-frame / hyperperiod concepts are noted but not
612// automatically computed here (TODO: LCM/hyperperiod + minor-frame packing).
613
614class CyclicExecutive {
615public:
616 // Catch-up policy for handling missed releases when `tick` is delayed.
617 enum class CatchUpPolicy {
618 // Call update for every missed activation (may starve others).
619 CatchAll,
620 // Call at most `max_catchups` updates per tick for a device, then
621 // advance its next-release to the first future release.
622 Limited,
623 // Do not call missed updates; advance next-release to the first
624 // release after the current time (misses are skipped).
625 SkipMisses
626 };

628 // Provide the device table (statically allocated by caller), count and
629 // a major frame duration in milliseconds (informational). The caller
630 // MUST provide a `next_release_buffer` with length >= `count` to avoid
631 // heap allocations. Returns true on success. The `policy` controls how
632 // missed activations are handled; `max_catchups` applies when policy is Limited.
633 static bool init(Device* devices, std::size_t count, uint32_t major_frame_ms,
634 uint64_t* next_release_buffer,
635 CatchUpPolicy policy = CatchUpPolicy::Limited,
636 uint32_t max_catchups = 1) {
637 if (!next_release_buffer) return false;
638 devices_ = devices;
639 count_ = count;
640 major_frame_ms_ = major_frame_ms;
641 current_time_ms_ = 0;
642 policy_ = policy;
643 max_catchups_ = max_catchups;
644
645 next_release_ = next_release_buffer;
646 for (std::size_t i = 0; i < count_; ++i) {
647 Device& d = devices_[i];
648 // Call device init if provided (framework-responsible init)
649 if (d.init) d.init(&d);
650 if (d.period_ms == 0) next_release_[i] = kNever;
651 else next_release_[i] = static_cast<uint64_t>(d.period_ms); // first release at period
652 }
653 return true;
654 }
655
656 // Reset timeline and reload next-release times into the provided buffer.
657 static void reset() {
658 current_time_ms_ = 0;
659 if (next_release_) {
660 for (std::size_t i = 0; i < count_; ++i) {
661 if (devices_[i].period_ms == 0) next_release_[i] = kNever;
662 else next_release_[i] = static_cast<uint64_t>(devices_[i].period_ms);
663 }
664 }
665 }
666
667 // Advance the executive by `delta_ms` on an absolute timeline. For each
668 // device we decrease its time-to-next-release and invoke `update` once
669 // per deadline that has passed (multiple firings allowed).
670 static void tick(uint32_t delta_ms) {
671 if (!devices_ || count_ == 0 || !next_release_) return;
672
673 current_time_ms_ += delta_ms;
674
675 for (std::size_t i = 0; i < count_; ++i) {
676 Device& d = devices_[i];
677 if (!d.update) continue;
678 if (d.period_ms == 0) continue;
679
680 // Determine how many activations are due
681 if (current_time_ms_ < next_release_[i]) continue;
682
683 uint64_t diff = current_time_ms_ - next_release_[i];
684 uint64_t total_due = diff / d.period_ms + 1; // at least one
685
686 switch (policy_) {
687 case CatchUpPolicy::CatchAll: {
688 for (uint64_t k = 0; k < total_due; ++k) d.update(&d);
689 // saturating add: avoid multiplication overflow
690 if (d.period_ms == 0) break;
691 uint64_t max_mul = UINT64_MAX / static_cast<uint64_t>(d.period_ms);
692 if (total_due > max_mul) {
693 next_release_[i] = kNever; // saturate
694 } else {
695 uint64_t incr = total_due * static_cast<uint64_t>(d.period_ms);
696 if (next_release_[i] > UINT64_MAX - incr) next_release_[i] = kNever;
697 else next_release_[i] += incr;
698 }
699 break;
700 }
701 case CatchUpPolicy::Limited: {
702 uint64_t calls = total_due;
703 if (calls > max_catchups_) calls = max_catchups_;
704 for (uint64_t k = 0; k < calls; ++k) d.update(&d);
705 // advance to the first release after current_time (skip remaining misses)
706 if (d.period_ms == 0) break;
707 uint64_t max_mul2 = UINT64_MAX / static_cast<uint64_t>(d.period_ms);
708 if (total_due > max_mul2) {
709 next_release_[i] = kNever;
710 } else {
711 uint64_t incr = total_due * static_cast<uint64_t>(d.period_ms);
712 if (next_release_[i] > UINT64_MAX - incr) next_release_[i] = kNever;
713 else next_release_[i] += incr;
714 }
715 break;
716 }
717 case CatchUpPolicy::SkipMisses: {
718 // do not call missed updates; schedule next after current time
719 if (d.period_ms == 0) break;
720 uint64_t max_mul3 = UINT64_MAX / static_cast<uint64_t>(d.period_ms);
721 if (total_due > max_mul3) {
722 next_release_[i] = kNever;
723 } else {
724 uint64_t incr = total_due * static_cast<uint64_t>(d.period_ms);
725 if (next_release_[i] > UINT64_MAX - incr) next_release_[i] = kNever;
726 else next_release_[i] += incr;
727 }
728 break;
729 }
730 }
731 }
732 }
733
734private:
735 // NOTE: this implementation uses global static state (single instance).
736 // For RTOS or multi-executive tests you may prefer an instance-based
737 // `CyclicExecutiveContext` that holds these fields; that allows multiple
738 // executives to coexist and simplifies unit testing. We keep the
739 // single-instance design for simplicity, but it is a documented
740 // design constraint.
741 static inline Device* devices_ = nullptr;
742 static inline std::size_t count_ = 0;
743 // Optional informative major frame (not used for wrapping here).
744 static inline uint32_t major_frame_ms_ = 0;
745 // Monotonic absolute time since init() in milliseconds
746 static inline uint64_t current_time_ms_ = 0;
747 // Per-device next absolute release time (ms). nullptr until init()
748 static inline uint64_t* next_release_ = nullptr;
749 static inline CatchUpPolicy policy_ = CatchUpPolicy::Limited;
750 static inline uint32_t max_catchups_ = 1;
751};
