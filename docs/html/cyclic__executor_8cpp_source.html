17static inline uint64_t sat_add_u64(uint64_t a, uint64_t b) noexcept {
18 if (a > UINT64_MAX - b) return UINT64_MAX;

23bool cyclic_init(ExecContext* ctx,
24 const Device* devices,
25 size_t device_count,
26 const ScheduleDef* schedules,
27 size_t schedule_count,
28 size_t initial_schedule,
29 Health* health_buffer,
30 DeviceState* state_buffer,
31 TimeSourceFn time_source,
32 OverrunPolicy overrun_policy) noexcept {
33 if (!ctx || !devices || device_count == 0 || !schedules || schedule_count == 0 || !state_buffer || !time_source) return false;
34 if (initial_schedule >= schedule_count) return false;
35
36 // Copy configuration into context (caller owns the pointed-to buffers)
37 ctx->devices = devices;
38 ctx->device_count = device_count;
39 ctx->schedules = schedules;
40 ctx->schedule_count = schedule_count;
41 ctx->active_schedule = initial_schedule;
42 ctx->pending_schedule = SIZE_MAX;
43 ctx->health = health_buffer; // optional: executor does not modify health here
44 ctx->states = state_buffer;
45 ctx->time_source = time_source;
46 ctx->overrun_policy = overrun_policy;
47
48 // Compute major-frame duration for the active schedule
49 const ScheduleDef& s = ctx->schedules[ctx->active_schedule];
50 uint64_t major = 0;
51 for (size_t i = 0; i < s.minor_count; ++i) {
52 major = sat_add_u64(major, s.minors[i].duration_us);
53 }
54 ctx->schedule_major_us = major;
55
56 // Initialize minor timing to now and set minor index to zero
57 ctx->minor_start_time_us = ctx->time_source();
58 ctx->current_minor = 0;
59 ctx->minor_used_us = 0;
60
61 // Initialize device states: call device init if present and mark Running
62 for (size_t i = 0; i < ctx->device_count; ++i) {
63 if (ctx->devices[i].init) ctx->devices[i].init(const_cast<Device*>(&ctx->devices[i]));
64 ctx->states[i] = DeviceState::Running;
65 }
66
67 return true;
68}

70static void execute_minor(ExecContext* ctx) noexcept {
71 const ScheduleDef& s = ctx->schedules[ctx->active_schedule];
72 if (ctx->current_minor >= s.minor_count) return;
73 const MinorFrameDef& minor = s.minors[ctx->current_minor];
74
75 uint64_t budget = minor.duration_us;
76 ctx->minor_used_us = 0;
77
78 for (size_t i = 0; i < minor.device_count; ++i) {
79 size_t dev_idx = minor.device_indices[i];
80 if (dev_idx >= ctx->device_count) continue;
81
82 // Only execute devices in Running state
83 if (ctx->states[dev_idx] != DeviceState::Running) continue;
84
85 uint64_t declared = ctx->devices[dev_idx].wcet_us;
86 uint64_t would = sat_add_u64(ctx->minor_used_us, declared);
87 if (would > budget) {
88 // budget exceeded — apply overrun policy deterministically
89 switch (ctx->overrun_policy) {
90 case OverrunPolicy::DropTask:
91 // skip this task, continue with next
92 continue;
93 case OverrunPolicy::SkipFrame:
94 // stop executing further tasks in this minor
95 ctx->minor_used_us = budget;
96 return;
97 case OverrunPolicy::SignalFault:
98 case OverrunPolicy::ResetSystem:
99 default:
100 // For policies that require system/fault hooks (not in scope here),
101 // conservatively stop executing remaining tasks in this minor.
102 ctx->minor_used_us = budget;
103 return;
104 }
105 }
106
107 // Execute the device update (executor does not measure runtime here)
108 if (ctx->devices[dev_idx].update) ctx->devices[dev_idx].update(const_cast<Device*>(&ctx->devices[dev_idx]));
109
110 // Deterministic accounting uses declared WCET (not measured)
111 ctx->minor_used_us = would;
112 if (ctx->minor_used_us >= budget) return;
113 }
114}

126void cyclic_tick_us(ExecContext* ctx, uint64_t current_time_us) noexcept {
127 if (!ctx || !ctx->schedules) return;
128
129 const ScheduleDef& s = ctx->schedules[ctx->active_schedule];
130 if (s.minor_count == 0) return;
131
132 // If current time hasn't reached the current minor end, do nothing.
133 uint64_t minor_end = sat_add_u64(ctx->minor_start_time_us, s.minors[ctx->current_minor].duration_us);
134 if (current_time_us < minor_end) return;
135
136 // Execute the minor whose deadline has passed.
137 execute_minor(ctx);
138
139 // Advance the minor start time deterministically by the minor duration we just executed.
140 ctx->minor_start_time_us = sat_add_u64(ctx->minor_start_time_us, s.minors[ctx->current_minor].duration_us);
141
142 // Move to next minor (wraps at major-frame boundary)
143 advance_minor(ctx);
144}
